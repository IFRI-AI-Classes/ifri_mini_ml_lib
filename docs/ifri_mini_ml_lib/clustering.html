 30        """
 31        Description:

 59    def _initialize_centroids(self, X):
 60        """
 61        Description:
 62        ------------
 63        Initializes centroids for KMeans clustering using the specified initialization method.
 64
 65        Arguments:
 66        -----------
 67        - X (ndarray): Input data array of shape (n_samples, n_features).
 68
 69        Functions:
 70        -----------
 71        - If 'random', selects k random samples as centroids.
 72        - If 'k-means++', selects centroids to spread out initial points.
 73        - Raises ValueError if init method is invalid.
 74
 75        Example:
 76        ---------
 77        self._initialize_centroids(X)
 78        """
 79        if self.random_state is not None:
 80            np.random.seed(self.random_state)
 81
 82        if self.init == 'random':
 83            indices = np.random.choice(X.shape[0], self.n_clusters, replace=False)
 84            self.centroids = X[indices]
 85        elif self.init == 'k-means++':
 86            self.centroids = [X[np.random.randint(X.shape[0])]]
 87            for _ in range(1, self.n_clusters):
 88                distances = np.array([min([euclidean_distance(c,x) for c in self.centroids]) for x in X])
 89                probabilities = distances / distances.sum()
 90                cumulative_probabilities = probabilities.cumsum()
 91                random_float = np.random.rand()
 92                for i, prob in enumerate(cumulative_probabilities):
 93                    if random_float < prob:
 94                        self.centroids.append(X[i])
 95                        break
 96            self.centroids = np.array(self.centroids)
 97        else:
 98            raise ValueError("init must be 'random' or 'k-means++'")

100    def fit(self, X):
101        """
102        Description:
103        ------------
104        Fits the KMeans model to data X by iteratively assigning points to clusters and updating centroids until convergence or max iterations.
105
106        Arguments:
107        -----------
108        - X (ndarray): Input data array of shape (n_samples, n_features).
109
110        Functions:
111        -----------
112        - Initializes centroids.
113        - Assigns each point to nearest centroid.
114        - Updates centroids as mean of assigned points.
115        - Stops when centroid shifts are below tolerance or max_iter reached.
116
117        Example:
118        ---------
119        kmeans.fit(X)
120        """
121        if self.random_state is not None:
122            np.random.seed(self.random_state)
123
124        self._initialize_centroids(X)
125
126        for _ in range(self.max_iter):
127            distances = np.array([[euclidean_distance(x, centroid) for centroid in self.centroids] for x in X])
128            self.labels = np.argmin(distances, axis=1)
129
130            new_centroids = np.array([X[self.labels == i].mean(axis=0) for i in range(self.n_clusters)])
131
132            if np.all(np.abs(new_centroids - self.centroids) < self.tol):
133                break
134
135            self.centroids = new_centroids

  6class DBSCAN:
  7    """
  8    Description:
  9    ------------
 10    DBSCAN (Density-Based Spatial Clustering of Applications with Noise) is a clustering algorithm that identifies clusters based on the density of points. It groups nearby points (defined by epsilon and min_samples) and labels outliers as noise.
 11
 12    Arguments:
 13        eps (float): The maximum radius to consider two points as neighbors.
 14        min_samples (int): The minimum number of points to form a cluster.
 15
 16    Example:
 17    >>> from ifri_mini_ml_lib.clustering import DBSCAN
 18    >>> dbscan = DBSCAN(eps=0.5, min_samples=5)
 19    >>> labels = dbscan.fit_predict(data)
 20    >>> dbscan.plot_clusters(data)
 21    """
 22
 23    def __init__(self, eps=0.5, min_samples=5):
 24        """
 25        Initializes the DBSCAN parameters.
 26
 27        Arguments:
 28            eps (float): The maximum radius to consider two points as neighbors.
 29            min_samples (int): The minimum number of points to form a cluster.
 30
 31        Example:
 32        >>> dbscan = DBSCAN(eps=0.5, min_samples=5)
 33        """
 34        self.eps = eps
 35        self.min_samples = min_samples
 36        self.labels = None  # Cluster labels
 37
 38    def fit_predict(self, data):
 39        """
 40        Performs DBSCAN clustering on the provided data.
 41
 42        Arguments:
 43            data (numpy.ndarray): The data to cluster (n_samples, n_features).
 44
 45        Example:
 46        >>> labels = dbscan.fit_predict(data)
 47        """
 48        self.labels = np.full(len(data), -1)  # Initialize all points as noise
 49        cluster_id = 0
 50
 51        for i in range(len(data)):
 52            if self.labels[i] != -1:
 53                continue  # Point already visited
 54
 55            # Find the neighbors of the current point
 56            neighbors = self._region_query(data, i)
 57
 58            if len(neighbors) < self.min_samples:
 59                # Not a central point, remains noise
 60                continue
 61
 62            # New cluster
 63            self._expand_cluster(data, i, cluster_id, neighbors)
 64            cluster_id += 1
 65
 66        return self.labels
 67
 68    def _region_query(self, data, point_index):
 69        """
 70        Finds the neighbors of a point within a given radius.
 71
 72        Arguments:
 73            data (numpy.ndarray): The data.
 74            point_index (int): The point index.
 75
 76        Example:
 77        >>> neighbors = self._region_query(data, 5)
 78        """
 79        neighbors = []
 80        for i in range(len(data)):
 81            if euclidean_distance(data[point_index], data[i]) <= self.eps:
 82                neighbors.append(i)
 83        return neighbors
 84        
 85    def _expand_cluster(self, data, point_index, cluster_id, neighbors):
 86        """
 87        Extends a cluster from a core point.
 88
 89        Arguments:
 90            data (numpy.ndarray): The data.
 91            point_index (int): The index of the core point.
 92            cluster_id (int): The ID of the current cluster.
 93            neighbors (list): The indices of the core point's neighbors.
 94
 95        Example:
 96        >>> self._expand_cluster(data, 10, 0, neighbors)
 97        """
 98        self.labels[point_index] = cluster_id
 99        i = 0
100        while i < len(neighbors):
101            neighbor_index = neighbors[i]
102            if self.labels[neighbor_index] == -1:
103                # Go to neighor
104                self.labels[neighbor_index] = cluster_id
105
106                # Find neighors to neighor
107                new_neighbors = self._region_query(data, neighbor_index)
108                if len(new_neighbors) >= self.min_samples:
109                    # Adds new neighbors to the list of neighbors to visit
110                    neighbors += set(new_neighbors)
111            i += 1
112
113    def plot_clusters(self, data):
114        """
115        Plots the resulting clusters after calling fit_predict().
116        Supports 1D, 2D and 3D data natively. For data with more than 3 features, applies PCA reduction to 3D.
117
118        Arguments:
119            data (numpy.ndarray): Input data (n_samples, n_features >= 1).
120
121        Example:
122        >>> dbscan.plot_clusters(data)
123        """
124        max_dimensions = 3
125        n_features = data.shape[1]
126
127        # PCA reduction if > 3D
128        if n_features > max_dimensions:
129            from ifri_mini_ml_lib.preprocessing.dimensionality_reduction.pca import PCA
130            reducer = PCA(n_component=3)
131            data = reducer.fit_transform(data)
132            print(f"Warning: PCA reduction applied from {n_features}D to {max_dimensions}D")
133            n_features = max_dimensions  # Update after PCA
134
135        unique_labels = set(self.labels)
136        colors = plt.colormaps.get_cmap('tab10')
137        fig = plt.figure(figsize=(10, 7))
138
139        # Case 1D
140        if n_features == 1:
141            ax = fig.add_subplot(111)
142            for label in unique_labels:
143                if label == -1:
144                    color = 'k'
145                    label_name = 'Noise'
146                else:
147                    color = colors(label)
148                    label_name = f'Cluster {label}'

 61    def _agglomerative_clustering(self, data):
 62        """
 63        Implements agglomerative (bottom-up) hierarchical clustering.
 64        """
 65        # Initialize each data point as a separate cluster (singleton)
 66        clusters = [{i} for i in range(len(data))]
 67    
 68        # Compute the pairwise distance matrix between all points
 69        distances = self._compute_distance_matrix(data)
 70    
 71        n_samples = len(data)
 72        # Track the current cluster indices (to assign unique indices to merged clusters)
 73        cluster_indices = list(range(n_samples))
 74        # Will store the linkage matrix: each row = [cluster1, cluster2, distance, new_cluster_size]
 75        linkage_matrix = []
 76        # Next cluster index to assign to a newly merged cluster
 77        next_cluster_idx = n_samples
 78
 79        # Main loop: continue until the desired number of clusters is reached
 80        while len(clusters) > self.n_clusters if self.n_clusters else len(clusters) > 1:
 81            # Find the two closest clusters
 82            min_i, min_j = None, None
 83            min_distance = float('inf')
 84            for i in range(len(clusters)):
 85                for j in range(i + 1, len(clusters)):
 86                    # Compute the linkage distance between clusters i and j
 87                    distance = self._compute_linkage_distance(clusters[i], clusters[j], distances, self.linkage)
 88                    if distance < min_distance:
 89                        min_distance = distance
 90                        min_i, min_j = i, j
 91
 92            # Record the merge in the linkage matrix
 93            c1_idx = cluster_indices[min_i]
 94            c2_idx = cluster_indices[min_j]
 95            new_cluster_size = len(clusters[min_i]) + len(clusters[min_j])
 96            linkage_matrix.append([c1_idx, c2_idx, min_distance, new_cluster_size])
 97
 98            # Merge the two closest clusters
 99            clusters[min_i] = clusters[min_i].union(clusters[min_j])
100            del clusters[min_j]
101            # Update cluster indices: assign a new index to the merged cluster, remove the merged one
102            cluster_indices[min_i] = next_cluster_idx
103            del cluster_indices[min_j]
104            next_cluster_idx += 1
105
106            # If the desired number of clusters is reached, exit the loop
107            if self.n_clusters and len(clusters) <= self.n_clusters:
108                break
109
110        # Store the linkage matrix as a numpy array (for dendrogram plotting)
111        self.linkage_matrix = np.array(linkage_matrix)
112
113        # Assign labels: for each cluster, assign the same label to all its points
114        labels = np.zeros(len(data), dtype=int)
115        for i, cluster in enumerate(clusters):
116            for point_index in cluster:
117                labels[point_index] = i
118
119        return labels

121    def _divisive_clustering(self, data, kmeans):
122        """
123        Implements divisive (top-down) hierarchical clustering.
124
125        Arguments:
126            data (numpy.ndarray): The dataset to be clustered.
127            kmeans (KMeans): An instance of a KMeans algorithm used to split clusters.
128
129        Returns:
130            numpy.ndarray: Cluster labels assigned to each data point.
131        """
132        # Initialization: all points are in a single cluster
133        clusters = [set(range(len(data)))]
134
135        while len(clusters) < self.n_clusters:
136            # Find the largest cluster (the one with the greatest number of points)
137            largest_cluster_index = np.argmax([len(cluster) for cluster in clusters])
138            largest_cluster = clusters[largest_cluster_index]
139
140            # Split the larger cluster into two sub-clusters
141            cluster1, cluster2 = self._bisect_cluster(data, largest_cluster, kmeans)
142
143            # Replace the original cluster with the two new subclusters
144            del clusters[largest_cluster_index]
145            clusters.append(cluster1)
146            clusters.append(cluster2)
147
148        # Assign labels
149        labels = np.zeros(len(data), dtype=int)
150        for i, cluster in enumerate(clusters):
151            for point_index in cluster:
152                labels[point_index] = i
153
154        return labels

156    def _bisect_cluster(self, data, cluster, kmeans):
157        """
158        Splits a cluster into two sub-clusters using a simple method (K-Means with k=2).
159
160        Arguments:
161            data (numpy.ndarray): The complete dataset.
162            cluster (set): Indices of the data points belonging to the cluster to be split.
163            kmeans (KMeans): An instance of the KMeans algorithm (can be reinitialized inside the method).
164
165        Returns:
166            tuple: Two sets representing the indices of the resulting sub-clusters.
167        """
168        # Convert cluster to data
169        cluster_data = data[list(cluster)]
170
171        # Use k-means to divide the cluster into two
172        kmeans = KMeans(n_clusters=2, random_state=42)  # Vous pouvez ajuster les paramÃ¨tres de KMeans
173        labels = kmeans.fit_predict(cluster_data)
174
175        # Create the two subclusters
176        cluster1 = set([list(cluster)[i] for i in range(len(cluster)) if labels[i] == 0])
177        cluster2 = set([list(cluster)[i] for i in range(len(cluster)) if labels[i] == 1])
178
179        return cluster1, cluster2

182        """
183        Description:
184        ------------

187        Arguments:
188        -----------
189        - data (numpy.ndarray): The dataset (n_samples, n_features).

193        - numpy.ndarray: A symmetric matrix containing pairwise distances between data points.

202    def _compute_linkage_distance(self, cluster1, cluster2, distances, linkage='single'):
203        """
204        Description:
205        ------------
206        Computes the distance between two clusters using the specified linkage criterion.
207
208        Arguments:
209        -----------
210        - cluster1 (set): Indices of the first cluster.
211        - cluster2 (set): Indices of the second cluster.
212        - distances (numpy.ndarray): Precomputed distance matrix between all data points.
213        - linkage (str): Linkage criterion to use ('single', 'complete', or 'average').
214
215        Returns:
216        --------
217        - float: The computed linkage distance between the two clusters.
218        """
219        if linkage == 'single':
220            # Minimum distance between points of the two clusters
221            min_distance = float('inf')
222            for i in cluster1:
223                for j in cluster2:
224                    distance = distances[i, j]
225                    if distance < min_distance:
226                        min_distance = distance
227            return min_distance
228        elif linkage == 'complete':
229            # Maximum distance between points of the two clusters
230            max_distance = float('-inf')
231            for i in cluster1:
232                for j in cluster2:
233                    distance = distances[i, j]
234                    if distance > max_distance:
235                        max_distance = distance
236            return max_distance
237        elif linkage == 'average':
238            # Average distance between points of the two clusters
239            total_distance = 0
240            for i in cluster1:
241                for j in cluster2:
242                    total_distance += distances[i, j]
243            return total_distance / (len(cluster1) * len(cluster2))
244        else:
245            raise ValueError("CritÃ¨re de linkage non reconnu. Choisissez 'single', 'complete' ou 'average'.")

248    def plot_dendrogram(self, labels=None):
249        """
250        Plots a dendrogram based on the linkage matrix constructed during hierarchical clustering.
251        The main branches corresponding to the final clusters are colored differently for better visualization.
252
253        Arguments:
254            data (numpy.ndarray): The dataset used for clustering.
255
256        Example:
257        >>> hierarchical = HierarchicalClustering(n_clusters=3, linkage='complete', method='agglomerative')
258        >>> labels = hierarchical.fit_predict(data)
259        >>> hierarchical.plot_dendrogram(data)
260        """
261        if self.method != 'agglomerative':
262            raise NotImplementedError("Dendrogram plotting is only implemented for agglomerative clustering.")
263
264        max_labels = 100  # Maximum number of labels to display on the leaves
265
266        # Check if the linkage matrix exists; if not, raise an error
267        if not hasattr(self, "linkage_matrix") or self.linkage_matrix is None:
268            raise ValueError("The linkage matrix is not available. Run fit_predict first.")
269
270        linkage_matrix = self.linkage_matrix
271        n_samples = linkage_matrix.shape[0] + 1
272
273        # If labels are not provided, use sample indices as labels
274        if labels is None:
275            labels = [str(i) for i in range(n_samples)]
276
277        # Option: show leaf labels only if there are not too many samples
278        show_labels = n_samples <= max_labels
279
280        # Determine the main clusters at the cut
281        # Take the n_clusters largest clusters formed at the end
282        n_clusters = self.n_clusters if self.n_clusters else 2
283        cluster_ids = list(range(n_samples, n_samples + linkage_matrix.shape[0]))
284        # The main clusters are those which remain at the end
285        main_clusters = set(cluster_ids[-n_clusters+1:]) if n_clusters > 1 else set([cluster_ids[-1]])
286
287        # Color palette for cluster branches
288        import matplotlib.cm as cm
289        colors = plt.colormaps.get_cmap('tab10')
290        branch_colors = {}
291
292        # Initialize horizontal (x) and vertical (y) positions for leaves
293        x_pos = {i: i for i in range(n_samples)}
294        y_pos = {i: 0 for i in range(n_samples)}
295
296        # Create the plot with a dynamic width depending on the number of samples
297        plt.figure(figsize=(max(8, min(20, n_samples // 5)), 6))
298
299        # Draw the branches of the dendrogram
300        for k, (c1, c2, dist, count) in enumerate(linkage_matrix):
301            c1, c2 = int(c1), int(c2)
302            # Protect against KeyError if a cluster index is missing
303            if c1 not in x_pos or c2 not in x_pos:
304                continue
305            x1, x2 = x_pos[c1], x_pos[c2]
306            y1, y2 = y_pos[c1], y_pos[c2]
307            x_new = (x1 + x2) / 2
308            cluster_idx = n_samples + k
309
310            # Color the main branches (clusters at the cut)
311            if cluster_idx in main_clusters:
312                color = colors(len(branch_colors) % colors.N)
313                branch_colors[cluster_idx] = color
314            else:
315                # Inherit the color from the parent branch if possible, otherwise use gray
316                color = branch_colors.get(c1, branch_colors.get(c2, "#888888"))
317
318            # Draw the vertical and horizontal lines for the current merge
319            plt.plot([x1, x1], [y1, dist], c=color)
320            plt.plot([x2, x2], [y2, dist], c=color)
321            plt.plot([x1, x2], [dist, dist], c=color)
322            # Store the position of the new cluster
323            x_pos[cluster_idx] = x_new
324            y_pos[cluster_idx] = dist
325
326        # Display leaf labels if not too many samples
327        if show_labels:
328            for i in range(n_samples):
329                plt.text(x_pos[i], -0.01, labels[i], ha='right', va='top', rotation=90, fontsize=8)
330
331        plt.xlabel("Sample index")
332        plt.ylabel("Distance")
333        plt.title("Dendrogram")
334        plt.tight_layout()
335        plt.show()
