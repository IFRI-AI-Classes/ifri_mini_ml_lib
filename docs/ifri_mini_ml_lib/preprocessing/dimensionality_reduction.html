 41    def fit(self, X: np.ndarray) -> 'PCA':
 42
 43        """
 44        Description:
 45            Model training (calculation of statistics and components).
 46
 47        Args:
 48            X (np.ndarray): 2D data matrix (shape: [n_samples, n_features])
 49
 50        Returns:
 51            PCA: Trained instance
 52
 53        Example:
 54            >>> data = np.array([[1, 2], [3, 4], [5, 6]])
 55            >>> model.fit(data)
 56        """
 57
 58        # Centering and normalization
 59        self.mean = np.mean(X, axis =  0)
 60        
 61        # Calculation of the covariance matrix
 62        self.cov = np.cov(X, rowvar=False) 
 63
 64        # Decomposition into eigenvalues ​​and eigenvectors
 65        self.eigen_values, eigen_vectors = np.linalg.eig(self.cov)
 66        
 67        # Sorts eigenvalues ​​and eigenvectors in descending order
 68        indices = np.argsort(self.eigen_values)[::-1]
 69        self.eigen_values = self.eigen_values[indices][:self.n_component]
 70        self.eigen_vectors = eigen_vectors[:, indices][:, :self.n_component]
 71
 72        # Select the first n eigenvectors
 73        self.components = self.eigen_vectors[:, :self.n_component].T
 74
 75        if self.components[0, 0] < 0:
 76            self.components = -self.components
 77
 78        return self

 65    def _euclidean_distance(self, X: np.ndarray) -> np.ndarray:
 66       
 67        """
 68        Description:
 69            Computes the squared Euclidean distance matrix.
 70
 71        Args:
 72            X (np.ndarray): Input data (shape: [n_samples, n_features])
 73
 74        Returns:
 75            np.ndarray: Squared distance matrix (shape: [n_samples, n_samples])
 76        """
 77
 78        sum_X = np.sum(np.square(X), axis=1)
 79        distances = np.add(-2 * np.dot(X, X.T), sum_X).T + sum_X
 80        np.fill_diagonal(distances, 0.0)
 81        return distances

 84    def _binary_search_perplexity(self, distances, perplexity, tol=1e-5, max_iter=50) -> np.ndarray:
 85
 86        """
 87        Description:
 88            Dichotomous search to find the optimal sigmas.
 89
 90        Args:
 91            distances (np.ndarray): Distance matrix
 92            perplexity (float): Target perplexity
 93            tol (float): Convergence tolerance
 94            max_iter (int): Maximum iterations
 95
 96        Returns:
 97            np.ndarray: Conditional probability matrix P
 98        """
 99
100        n_samples = distances.shape[0]
101        P = np.zeros((n_samples, n_samples))
102        beta = np.ones((n_samples, 1))
103        log_perplexity = np.log(perplexity)
104        
105        # We ignore the diagonal (distance from oneself = 0)
106        for i in range(n_samples):
107            beta_min = -np.inf
108            beta_max = np.inf
109            dist_i = distances[i, np.concatenate((np.r_[0:i], np.r_[i+1:n_samples]))]
110            
111            for _ in range(max_iter):
112                # Calculation of conditional probabilities
113                P_i = np.exp(-dist_i * beta[i])
114                sum_Pi = np.sum(P_i)
115                
116                if sum_Pi == 0:
117                    sum_Pi = 1e-8
118                
119                # Calculation of entropy
120                H = np.log(sum_Pi) + beta[i] * np.sum(dist_i * P_i) / sum_Pi
121                P_i = P_i / sum_Pi
122                
123                # Beta adjustment (binary precision)
124                H_diff = H - log_perplexity
125                if np.abs(H_diff) < tol:
126                    break
127                
128                if H_diff > 0:
129                    beta_min = beta[i].copy()
130                    if beta_max == np.inf:
131                        beta[i] *= 2.0
132                    else:
133                        beta[i] = (beta[i] + beta_max) / 2.0
134                else:
135                    beta_max = beta[i].copy()
136                    if beta_min == -np.inf:
137                        beta[i] /= 2.0
138                    else:
139                        beta[i] = (beta[i] + beta_min) / 2.0
140            
141            # Fill in the P matrix
142            P[i, np.concatenate((np.r_[0:i], np.r_[i+1:n_samples]))] = P_i
143        
144        return P

147    def _compute_joint_probabilities(self, X: np.ndarray, perplexity: float) -> np.ndarray:
148
149        """
150        Description:
151            Computes the joint probabilities P in high dimensions.
152
153        Args:
154            X (np.ndarray): Original data
155            perplexity (float): Desired perplexity
156
157        Returns:
158            np.ndarray: Symmetric joint probability matrix
159        """
160
161        # Calculation of square Euclidean distances
162        distances = self._euclidean_distance(X)
163        
164        # Calculation of conditional probabilities
165        P = self._binary_search_perplexity(distances, perplexity)
166        
167        # Symmetrization and normalization
168        P = (P + P.T) / (2.0 * P.shape[0])
169        P /= np.sum(P)
170        P = np.maximum(P, 1e-12)
171        
172        return P

174    def _compute_low_dimensional_probabilities(self, Y: np.ndarray) -> np.ndarray:
175
176        """
177        Description:
178            Computes the low-dimensional joint probabilities q_ij according to a Student t distribution.
179
180        Args:
181            Y (np.ndarray): Current low-dimensional embedding (shape: [n_samples, n_components])
182
183        Returns:
184            np.ndarray: Probability matrix Q (shape: [n_samples, n_samples])
185
186        Example:
187            >>> q_matrix = self._compute_low_dimensional_probabilities(current_embedding)
188        """
189
190        distances = self._euclidean_distance(Y)
191        inv_distances = 1.0 / (1.0 + distances)
192        np.fill_diagonal(inv_distances, 0.0)
193        Q = inv_distances / np.sum(inv_distances)
194        Q = np.maximum(Q, 1e-12)
195        return Q

198    def _compute_gradient(self, P: np.ndarray, Q: np.ndarray, Y: np.ndarray) -> np.ndarray:
199
200        """
201        Description:
202            Computes the gradient of the KL divergence with respect to the embedding coordinates.
203
204        Args:
205            P (np.ndarray): High-dimensional probability matrix
206            Q (np.ndarray): Low-dimensional probability matrix
207            Y (np.ndarray): Current embedding
208
209        Returns:
210            np.ndarray: Gradient matrix (same shape as Y)
211
212        Example:
213            >>> grad = self._compute_gradient(p_matrix, q_matrix, current_embedding)
214        """
215
216        diff = Y[:, None, :] - Y[None, :, :]  # Shape: (n_samples, n_samples, n_components)
217        pq_diff = (P - Q)[..., None] * (1.0 / (1.0 + np.clip(self._euclidean_distance(Y), 1e-8, None)))[..., None] 
218        
219        return 4 * (pq_diff * diff).sum(axis=1)  # Shape: (n_samples, n_components)
220

222    def _compute_kl_divergence(self, P: np.ndarray, Q: np.ndarray) -> float:
223
224        """
225        Description:
226            Computes the Kullback-Leibler divergence between the P and Q distributions.
227
228        Args:
229            P (np.ndarray): Reference probability distribution
230            Q (np.ndarray): Approximate distribution
231
232        Returns:
233            float: KL divergence value (in bits)
234
235        Example:
236            >>> kl = self._compute_kl_divergence(p_matrix, q_matrix)
237        """
238        return np.sum(P * np.log(P / Q))

241    def fit(self, X: np.ndarray) -> 'TSNE':
242
243        """
244        Description:
245            Training the t-SNE model.
246
247        Args:
248            X (np.ndarray): Data to project (shape: [n_samples, n_features])
249
250        Returns:
251            TSNE: Trained instance
252
253        Raises:
254            ValueError: If n_samples < 3 * perplexity
255
256        Example:
257            >>> data = np.random.rand(100, 10)
258            >>> tsne.fit(data)
259        """
260
261        n_samples = X.shape[0]
262        
263        # Data verification
264        if n_samples < 3 * self.perplexity:
265            raise ValueError(f"The number of samples ({n_samples}) must be at least 3 * perplexity ({3*self.perplexity})")
266        
267        if self.verbose:
268            print("Calculation of joint probabilities P...")
269        
270        # Calculation of P in high dimension
271        P = self._compute_joint_probabilities(X, self.perplexity)
272        P_early = P * self.early_exaggeration
273        
274        if self.init == 'random':
275            # Random initialization of Y
276            Y = 1e-4 * np.random.randn(n_samples, self.n_components).astype(np.float32)
277        else:
278            # Initialization of Y by pca
279            Y = PCA(n_component=self.n_components).fit_transform(X)
280            Y = Y / np.std(Y, axis=0) * 1e-4  
281        
282        exaggeration_end_iter = 250
283
284        # Variables for optimization
285        velocity = np.zeros_like(Y)      # Initialization of momentum
286        gains = np.ones_like(Y)
287        
288        if self.verbose:
289            print("Optimization of embedding...")
290        
291        # Optimisation
292        for i in range(self.n_iter):
293            # Calculation of Q in low dimension
294            Q = self._compute_low_dimensional_probabilities(Y)
295            
296            # Gradient calculation
297            gradient = self._compute_gradient(P_early if i < exaggeration_end_iter else P , Q, Y)
298            grad_norm = np.linalg.norm(gradient)
299            
300            # Update with momentum
301            momentum = 0.5 if i < exaggeration_end_iter else 0.8
302
303            # Update of gains
304            gains = (gains + 0.2) * ((gradient > 0) != (velocity > 0)) + \
305                    (gains * 0.8) * ((gradient > 0) == (velocity > 0))
306            gains = np.clip(gains, 0.01, np.inf)
307
308            # velocity's update with momentum
309            velocity = momentum * velocity - self.learning_rate * (gains * gradient)
310            
311            # Update of embedding
312            Y += velocity
313

241    def fit(self, X: np.ndarray) -> 'TSNE':
242
243        """
244        Description:
245            Training the t-SNE model.
246
247        Args:
248            X (np.ndarray): Data to project (shape: [n_samples, n_features])
249
250        Returns:
251            TSNE: Trained instance
252
253        Raises:
254            ValueError: If n_samples < 3 * perplexity
255
256        Example:
257            >>> data = np.random.rand(100, 10)
258            >>> tsne.fit(data)
259        """
260
261        n_samples = X.shape[0]
262        
263        # Data verification
264        if n_samples < 3 * self.perplexity:
265            raise ValueError(f"The number of samples ({n_samples}) must be at least 3 * perplexity ({3*self.perplexity})")
266        
267        if self.verbose:
268            print("Calculation of joint probabilities P...")
269        
270        # Calculation of P in high dimension
271        P = self._compute_joint_probabilities(X, self.perplexity)
272        P_early = P * self.early_exaggeration
273        
274        if self.init == 'random':
275            # Random initialization of Y
276            Y = 1e-4 * np.random.randn(n_samples, self.n_components).astype(np.float32)
277        else:
278            # Initialization of Y by pca
279            Y = PCA(n_component=self.n_components).fit_transform(X)
280            Y = Y / np.std(Y, axis=0) * 1e-4  
281        
282        exaggeration_end_iter = 250
283
284        # Variables for optimization
285        velocity = np.zeros_like(Y)      # Initialization of momentum
286        gains = np.ones_like(Y)
287        
288        if self.verbose:
289            print("Optimization of embedding...")
290        
291        # Optimisation
292        for i in range(self.n_iter):
293            # Calculation of Q in low dimension
294            Q = self._compute_low_dimensional_probabilities(Y)
295            
296            # Gradient calculation
297            gradient = self._compute_gradient(P_early if i < exaggeration_end_iter else P , Q, Y)
298            grad_norm = np.linalg.norm(gradient)
299            
300            # Update with momentum
301            momentum = 0.5 if i < exaggeration_end_iter else 0.8
302
303            # Update of gains
304            gains = (gains + 0.2) * ((gradient > 0) != (velocity > 0)) + \
305                    (gains * 0.8) * ((gradient > 0) == (velocity > 0))
306            gains = np.clip(gains, 0.01, np.inf)
307
308            # velocity's update with momentum
309            velocity = momentum * velocity - self.learning_rate * (gains * gradient)
310            
311            # Update of embedding
312            Y += velocity
313
314            # Data Centering
315            Y -= Y.mean(axis=0)
316
317            # Dynamic learning rate update
318            if i == exaggeration_end_iter:
319                self.learning_rate *= 0.8
320            
321            # Reinitialization of P after early exaggeration
322            if i == exaggeration_end_iter:
323                P = self._compute_joint_probabilities(X, self.perplexity)
324            
325            # Calculation of KL divergence
326            kl_div = self._compute_kl_divergence(P_early if i < exaggeration_end_iter else P, Q)
327            
328            # Displaying information
329            if self.verbose >= 1 and i % 100 == 0:
330                print(f"Iteration {i}: KL divergence = {kl_div:.4f}, Gradient norm = {grad_norm:.4f}, LR={self.learning_rate:.1f}")
331                
332            if grad_norm < self.min_grad_norm:
333                if self.verbose:
334                    print(f"Premature stop at iteration {i}: gradient norm too low")
335                break
336        
337        # Saving results
338        self.embedding_ = Y.copy()
339        #self.embedding_ = (Y - np.mean(Y, axis=0)) / np.std(Y, axis=0)  # Normalization
340        self.kl_divergence_ = kl_div
341        self.n_iter_ = i + 1
342        
343        return self
