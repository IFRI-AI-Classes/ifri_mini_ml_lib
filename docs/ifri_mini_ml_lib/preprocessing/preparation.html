 12class MissingValueHandler:
 13    """
 14    Class to handle missing values in datasets using various techniques
 15    (deletion, statistical imputation, default value, KNN, and regression).
 16    """
 17
 18    def __init__(self):
 19        pass
 20
 21    def _convert_to_dataframe(self, X):
 22        """
 23        Description:
 24            Converts input to a pandas DataFrame if not already one.
 25
 26        Args:
 27            X (pd.DataFrame | np.ndarray | array-like): Input data.
 28
 29        Returns:
 30            pd.DataFrame: Data converted to DataFrame.
 31
 32        Example:
 33            >>> handler._convert_to_dataframe(np.array([[1, 2], [3, np.nan]]))
 34        """
 35        if isinstance(X, pd.DataFrame):
 36            return X
 37        return pd.DataFrame(X)
 38
 39    def remove_missing(self, X, threshold=0.5, axis=0):
 40        """
 41        Description:
 42            Removes rows or columns containing too many missing values.
 43
 44        Args:
 45            X (pd.DataFrame | np.ndarray): Input data.
 46            threshold (float): Proportion threshold of NaN allowed before deletion (between 0 and 1).
 47            axis (int): Axis to process (0 for rows, 1 for columns).
 48
 49        Returns:
 50            pd.DataFrame: Data after deletion.
 51
 52        Example:
 53            >>> handler.remove_missing(df, threshold=0.4, axis=0)
 54        """
 55        df = self._convert_to_dataframe(X)
 56        df_cleaned = df.dropna(thresh=int(threshold * df.shape[axis]), axis=axis)
 57        return df_cleaned
 58
 59    def impute_statistical(self, X, strategy="mean"):
 60        """
 61        Description:
 62            Imputes missing values using a statistical strategy.
 63
 64        Args:
 65            X (pd.DataFrame | np.ndarray): Input data.
 66            strategy (str): Imputation method ('mean', 'median' or 'mode').
 67
 68        Returns:
 69            pd.DataFrame: Imputed data.
 70
 71        Example:
 72            >>> handler.impute_statistical(df, strategy="median")
 73        """
 74        df = self._convert_to_dataframe(X)
 75        df = df.copy()  # Create a copy to avoid modifying the original DataFrame in-place
 76    
 77        for col in df.columns:
 78            if df[col].isnull().sum() > 0:
 79                if strategy == "mean":
 80                    df[col] = df[col].fillna(df[col].mean())
 81                elif strategy == "median":
 82                    df[col] = df[col].fillna(df[col].median())
 83                elif strategy == "mode":
 84                    df[col] = df[col].fillna(df[col].mode()[0])
 85        
 86        return df
 87
 88    def impute_default(self, X, value=0):
 89        """
 90        Description:
 91            Replaces all missing values with a constant value.
 92
 93        Args:
 94            X (pd.DataFrame | np.ndarray): Input data.
 95            value (int | float): Value used to replace NaN.
 96
 97        Returns:
 98            pd.DataFrame: Imputed data.
 99
100        Example:
101            >>> handler.impute_default(df, value=-1)
102        """
103        df = self._convert_to_dataframe(X)
104        return df.fillna(value)

106    def impute_knn(self, X, k=3, task='regression'):
107        """
108        Description:
109            Imputes missing values using k-nearest neighbors.
110
111        Args:
112            X (pd.DataFrame | np.ndarray): Input data.
113            k (int): Number of neighbors to consider.
114
115        Returns:
116            pd.DataFrame: Imputed data.
117
118        Example:
119            >>> handler.impute_knn(df, k=5)
120        """
121
122        df = self._convert_to_dataframe(X)
123
124        for col in df.columns:
125            if df[col].isnull().sum() > 0:
126                known = df[df[col].notnull()].fillna(0)
127                unknown = df[df[col].isnull()].fillna(0)
128                
129                if known.empty:
130                    continue
131                
132                X_train = known.drop(col, axis=1).values
133                y_train = known[col].values
134                X_test = unknown.drop(col, axis=1).values
135                
136                knn = KNN(k=k, task=task)
137                knn.fit(X_train, y_train)
138                predictions = knn.predict(X_test)
139                
140                df.loc[unknown.index, col] = predictions
141
142        return df

144    def impute_regression(self, X, target_col):
145        """
146        Description:
147            Imputes a target column by predicting it with a regression model
148            based on other columns.
149
150        Args:
151            X (pd.DataFrame | np.ndarray): Input data.
152            target_col (str): Name of column to impute.
153
154        Returns:
155            pd.DataFrame: Imputed data.

162        """
163        df = self._convert_to_dataframe(X)

165        if target_col not in df.columns:
166            raise ValueError(f"La colonne '{target_col}' n'existe pas dans X.")
167
168        known = df[df[target_col].notnull()]
169        unknown = df[df[target_col].isnull()]
170
171        if len(unknown) == 0:
172            return df
173
174
175        model = LinearRegression()
176        model.fit(known.drop(target_col, axis=1).values, known[target_col].values)
177        predicted_values = model.predict(unknown.drop(target_col, axis=1).values)

179        df.loc[df[target_col].isnull(), target_col] = predicted_values
180        return df

class CategoricalEncoder: 
  4class CategoricalEncoder:
  5    """
  6    A flexible categorical encoder that supports multiple encoding techniques for categorical variables.
  7    
  8    Description:
  9        This class provides functionality to encode categorical variables using various techniques:
 10        - Label Encoding: Assigns each unique category an integer value
 11        - Ordinal Encoding: Similar to label encoding but categories are sorted first
 12        - Frequency Encoding: Replaces categories with their frequency in the dataset
 13        - Target Encoding: Replaces categories with the mean of the target variable for that category
 14        - One-Hot Encoding: Creates binary columns for each category
 15        
 16    Args:
 17        encoding_type (str): Type of encoding to apply. Options: 'onehot', 'label', 'ordinal', 'frequency', 'target'.
 18                            Default is 'onehot'.
 19        target_column (str): Name of the target column (required for target encoding). Default is None.
 20    """
 21    
 22    def __init__(self, encoding_type='onehot', target_column=None):
 23        self.encoding_type = encoding_type
 24        self.target_column = target_column
 25        self.mapping = {}  # Stores encoding mappings
 26
 27    def fit(self, X, y=None):
 28        """
 29        Learn the encoding mappings from the data.
 30        
 31        Description:
 32            Computes and stores the necessary encoding information based on the training data.
 33            
 34        Args:
 35            X (pd.DataFrame): Input data containing categorical features to encode
 36            y (pd.Series, optional): Target variable (required for target encoding)
 37            
 38        Raises:
 39            ValueError: If target encoding is selected but no target variable is provided
 40        """
 41        if self.encoding_type == 'target' and y is None:
 42            raise ValueError("Target encoding requires target column `y`.")
 43        
 44        for column in X.select_dtypes(include='object').columns:
 45            if self.encoding_type == 'label':
 46                self.mapping[column] = {cat: idx for idx, cat in enumerate(X[column].unique())}
 47            elif self.encoding_type == 'ordinal':
 48                if column == 'size':
 49                    categories = ['S', 'M', 'L']
 50                    self.mapping[column] = {cat: idx for idx, cat in enumerate(categories)}
 51                else:
 52                    self.mapping[column] = {cat: idx for idx, cat in enumerate(sorted(X[column].unique()))}
 53            elif self.encoding_type == 'frequency':
 54                freq = X[column].value_counts(normalize=True)
 55                self.mapping[column] = freq.to_dict()
 56            elif self.encoding_type == 'target':
 57                target_means = X.join(y).groupby(column)[self.target_column].mean()
 58                self.mapping[column] = target_means.to_dict()
 59            elif self.encoding_type == 'onehot':
 60                # No mapping needed for one-hot encoding
 61                pass
 62            else:
 63                raise ValueError(f"Unknown encoding type: {self.encoding_type}")
 64
 65    def transform(self, X):
 66        """
 67        Apply the encoding to new data using the learned mappings.
 68        
 69        Description:
 70            Transforms the input data by applying the encoding learned during fit().
 71            
 72        Args:
 73            X (pd.DataFrame): Data to be encoded
 74            
 75        Returns:
 76            pd.DataFrame: Transformed data with categorical features encoded according to the specified method
 77        """
 78        X_encoded = X.copy()
 79
 80        for column in X_encoded.select_dtypes(include='object').columns:
 81            if self.encoding_type in ['label', 'ordinal', 'frequency', 'target']:
 82                X_encoded[column] = X_encoded[column].map(self.mapping[column])
 83            elif self.encoding_type == 'onehot':
 84                dummies = pd.get_dummies(X_encoded[column], prefix=column)
 85                X_encoded = X_encoded.drop(column, axis=1)
 86                X_encoded = pd.concat([X_encoded, dummies], axis=1)
 87
 88        return X_encoded
 89
 90    def fit_transform(self, X, y=None):
 91        """
 92        Learn the encoding and apply it to the training data in one step.
 93        
 94        Description:
 95            Convenience method that combines fit() and transform() operations.
 96

 98            X (pd.DataFrame): Training data to fit and transform
 99            y (pd.Series, optional): Target variable (required for target encoding)
100            
101        Returns:
102            pd.DataFrame: Transformed data with categorical features encoded
103        """
104        self.fit(X, y)
105        return self.transform(X)

 16    def train_test_split(self, X, y=None, test_size=0.2, val_set=False, val_size=0.2):
 17        """
 18        Split data randomly into train and test sets, with optional validation set.
 19        
 20        Args:
 21            X (pd.DataFrame): Feature data
 22            y (pd.Series, optional): Target data. Defaults to None.
 23            test_size (float, optional): Proportion of dataset to include in test split. Defaults to 0.2.
 24            val_set (bool, optional): Whether to create validation set. Defaults to False.
 25            val_size (float, optional): Proportion of train set to include in validation split. Defaults to 0.2.
 26            
 27        Returns:
 28            If y is None:
 29                X_train, X_test or X_train, X_val, X_test if val_set=True
 30            If y is provided:
 31                X_train, X_test, y_train, y_test or 
 32                X_train, X_val, X_test, y_train, y_val, y_test if val_set=True
 33                
 34        Example:
 35            >>> splitter = DataSplitter(seed=42)
 36            >>> X_train, X_test = splitter.train_test_split(X)
 37            >>> X_train, X_val, X_test, y_train, y_val, y_test = splitter.train_test_split(
 38            ...     X, y, test_size=0.3, val_set=True, val_size=0.1)
 39        """
 40        indices = np.arange(len(X))
 41        np.random.shuffle(indices)
 42        
 43        test_end = int(test_size * len(X))
 44        test_idx = indices[:test_end]
 45        train_idx = indices[test_end:]
 46        
 47        if not val_set:
 48            if y is None:
 49                return X.iloc[train_idx], X.iloc[test_idx]
 50            else:
 51                return X.iloc[train_idx], X.iloc[test_idx], y.iloc[train_idx], y.iloc[test_idx]
 52        else:
 53            val_split = int(val_size * len(train_idx))
 54            val_idx = train_idx[:val_split]
 55            new_train_idx = train_idx[val_split:]
 56            
 57            if y is None:
 58                return X.iloc[new_train_idx], X.iloc[val_idx], X.iloc[test_idx]
 59            else:
 60                return (X.iloc[new_train_idx], X.iloc[val_idx], X.iloc[test_idx],
 61                        y.iloc[new_train_idx], y.iloc[val_idx], y.iloc[test_idx])

 63    def stratified_train_test_split(self, X, y, test_size=0.2, val_set=False, val_size=0.2):
 64        """
 65        Split data into train and test sets preserving class distribution, with optional validation set.
 66        
 67        Args:
 68            X (pd.DataFrame): Feature data
 69            y (pd.Series): Target data (required for stratified split)
 70            test_size (float, optional): Proportion of dataset to include in test split. Defaults to 0.2.
 71            val_set (bool, optional): Whether to create validation set. Defaults to False.
 72            val_size (float, optional): Proportion of train set to include in validation split. Defaults to 0.2.
 73            
 74        Returns:
 75            If val_set=False:
 76                X_train, X_test, y_train, y_test
 77            If val_set=True:
 78                X_train, X_val, X_test, y_train, y_val, y_test
 79                
 80        Example:
 81            >>> splitter = DataSplitter(seed=42)
 82            >>> X_train, X_test, y_train, y_test = splitter.stratified_train_test_split(X, y)
 83            >>> X_train, X_val, X_test, y_train, y_val, y_test = splitter.stratified_train_test_split(
 84            ...     X, y, val_set=True, val_size=0.1)
 85        """
 86        assert y is not None, "Stratified split requires target vector y"
 87        
 88        classes = np.unique(y)
 89        train_idx, val_idx, test_idx = [], [], []
 90        
 91        for cls in classes:
 92            cls_indices = np.where(y == cls)[0]
 93            np.random.shuffle(cls_indices)
 94            n = len(cls_indices)
 95            
 96            test_end = int(test_size * n)
 97            cls_test = cls_indices[:test_end]
 98            cls_train = cls_indices[test_end:]
 99            
100            if val_set:
101                val_split = int(val_size * len(cls_train))
102                cls_val = cls_train[:val_split]
103                cls_train = cls_train[val_split:]
104                val_idx.extend(cls_val)
105            
106            test_idx.extend(cls_test)
107            train_idx.extend(cls_train)
108        
109        if not val_set:
110            return X.iloc[train_idx], X.iloc[test_idx], y.iloc[train_idx], y.iloc[test_idx]
111        else:
112            return (X.iloc[train_idx], X.iloc[val_idx], X.iloc[test_idx],
113                    y.iloc[train_idx], y.iloc[val_idx], y.iloc[test_idx])

115    def temporal_train_test_split(self, X, y=None, test_size=0.2, val_set=False, val_size=0.2):
116        """
117        Split time series data into train and test sets (and optionally validation),
118        preserving temporal order.
119        
120        Args:
121            X (pd.DataFrame): Feature data (time-ordered)
122            y (pd.Series, optional): Target data (time-ordered)
123            test_size (float, optional): Proportion of dataset to include in test split. Defaults to 0.2.
124            val_set (bool, optional): Whether to create validation set. Defaults to False.
125            val_size (float, optional): Proportion of train set to include in validation split. Defaults to 0.2.
126            
127        Returns:
128            If y is None:
129                X_train, X_test or X_train, X_val, X_test if val_set=True
130            If y is provided:
131                X_train, X_test, y_train, y_test or 
132                X_train, X_val, X_test, y_train, y_val, y_test if val_set=True
133                
134        Example:
135            >>> splitter = DataSplitter()
136            >>> X_train, X_test = splitter.temporal_train_test_split(X)
137            >>> X_train, X_val, X_test, y_train, y_val, y_test = splitter.temporal_train_test_split(
138            ...     X, y, val_set=True)
139        """
140        n_samples = len(X)
141        test_end = int(test_size * n_samples)
142        
143        X_test = X.iloc[-test_end:]  # Take most recent samples for test
144        X_train = X.iloc[:-test_end]
145        
146        if y is None:
147            if not val_set:
148                return X_train, X_test
149            else:
150                val_end = int(val_size * len(X_train))
151                X_val = X_train.iloc[-val_end:]
152                X_train = X_train.iloc[:-val_end]
153                return X_train, X_val, X_test
154        else:
155            y_test = y.iloc[-test_end:]
156            y_train = y.iloc[:-test_end]
157
