 52    def _fit_simple(self, X: np.ndarray, y: np.ndarray) -> 'LinearRegression':
 53        """
 54        Internal method to train simple linear regression model.
 55        
 56        Args:
 57            X (np.ndarray): 1D array of shape [n_samples]
 58            y (np.ndarray): 1D array of shape [n_samples]
 59
 60        Returns:
 61            self: Trained model instance
 62
 63        Raises:
 64            ValueError: If division by zero occurs in least squares
 65
 66        Example:
 67            >>> model = LinearRegression()
 68            >>> model._fit_simple(np.array([1,2,3]), np.array([2,4,6]))
 69        """
 70        m = len(X)
 71        if m == 0:
 72            raise ValueError("Input datas are empties")
 73
 74        X_mean = np.mean(X)
 75        y_mean = np.mean(y)
 76
 77        if self.method == "least_squares":
 78            num = np.sum((X - X_mean) * (y - y_mean))
 79            den = np.sum((X - X_mean) ** 2)
 80            if abs(float(den)) < 1e-10:
 81                raise ValueError("Division par zÃ©ro dÃ©tectÃ©e dans les moindres carrÃ©s")
 82            self.w = num / den
 83            self.b = y_mean - self.w * X_mean
 84
 85        elif self.method == "gradient_descent":
 86            self.w = 0.0
 87            self.b = 0.0
 88            for _ in range(self.epochs):
 89                y_pred = self.w * X + self.b

 91                db = (-2 / m) * np.sum(y - y_pred)
 92                self.w -= self.learning_rate * dw
 93                self.b -= self.learning_rate * db
 94        else:
 95            raise ValueError(f"MÃ©thode inconnue: {self.method}")

 98    def _fit_multiple(self, X: np.ndarray, y: np.ndarray) -> 'LinearRegression':
 99        """
100        Internal method to train multiple linear regression model.
101        
102        Args:
103            X (np.ndarray): 2D array of shape [n_samples, n_features]
104            y (np.ndarray): 1D array of shape [n_samples]
105
106        Returns:
107            self: Trained model instance
108
109        Example:
110            >>> model = LinearRegression()
111            >>> model._fit_multiple(np.array([[1],[2],[3]]), np.array([1,2,3]))
112        """
113        m, n = X.shape
114        X_b = np.hstack([np.ones((m, 1)), X])
115        if self.method == "least_squares":
116            theta = np.linalg.pinv(X_b.T @ X_b) @ X_b.T @ y.reshape(-1, 1)
117            self.b = float(theta[0].item())
118            self.w = theta[1:].flatten()
119        elif self.method == "gradient_descent":
120            self.w = np.zeros(n)
121            self.b = 0.0
122            for _ in range(self.epochs):
123                y_pred = X @ self.w + self.b
124                error = y - y_pred
125                dw = (-2 / m) * (X.T @ error)
126                db = (-2 / m) * np.sum(error)
127                self.w -= self.learning_rate * dw
128                self.b -= self.learning_rate * db
129        else:
130            raise ValueError(f"MÃ©thode inconnue: {self.method}")
131        return self

 24    def _polynomial_features(self, X: np.ndarray) -> np.ndarray:
 25        """
 26        Generate polynomial features from input data including cross-terms.
 27        
 28        Args:
 29            X (np.ndarray): Input data of shape [n_samples, n_features]
 30
 31        Returns:
 32            np.ndarray: Polynomial features matrix including all terms up to the specified degree
 33        
 34        Example: degree=2 et X=[[a, b]] :
 35        Output : [a, b, aÂ², ab, bÂ²]
 36        """
 37        # Ensure X is properly formatted as a 2D array
 38        if not isinstance(X[0], (list, np.ndarray)):
 39            X = [[x] for x in X]
 40        
 41        X_array = np.array(X)
 42        n_samples, n_features = X_array.shape
 43        
 44        # For a single feature, just compute powers
 45        if n_features == 1:
 46            poly_features = []
 47            for row in X_array:
 48                features = [row[0] ** p for p in range(1, self.degree + 1)]
 49                poly_features.append(features)
 50            return np.array(poly_features)
 51        
 52        # For multiple features, we need to include cross-terms
 53        from itertools import combinations_with_replacement
 54        
 55        poly_features = []
 56        for sample in X_array:
 57            sample_features = []
 58            # Generate all combinations of features up to the degree
 59            for d in range(1, self.degree + 1):
 60                for combination in combinations_with_replacement(range(n_features), d):
 61                    # Compute the term (product of features raised to powers)
 62                    term = 1
 63                    for feature_idx in combination:
 64                        term *= sample[feature_idx]
 65                    sample_features.append(term)
 66            poly_features.append(sample_features)
 67        
 68        return np.array(poly_features)

111        """
112        Computes the Radial Basis Function (RBF) kernel.
113

154    def fit(self, X: np.ndarray, Y: np.ndarray) -> None:
155        """
156        Trains the SVR model using convex optimization.
157
158        Args:
159            X (ndarray): Training features.
160            Y (ndarray): Training targets.
161
162        Returns:
163            None
164
165        Example:
166            model.fit(X_train, y_train)
167        """
168        self.train_X = X
169        self.train_y = Y
170
171        K = self.get_kernel()(self.train_X)
172        K_reg = K + 1e-6 * np.eye(K.shape[0])
173        n = self.train_X.shape[0]
174    
175        alpha = cp.Variable(n)
176        alpha_star = cp.Variable(n)
177    
178        objective = cp.Minimize(
179            0.5 * cp.quad_form(alpha - alpha_star, cp.psd_wrap(K_reg))
180            + self.eps * cp.sum(alpha + alpha_star)
181            - cp.sum(self.train_y.T @ (alpha - alpha_star))
182        )
183    
184        constraints = [
185            cp.sum(alpha - alpha_star) == 0,
186            alpha >= 0,
187            alpha_star >= 0,
188            alpha <= self._c_reg,
189            alpha_star <= self._c_reg
190        ]
191    
192        problem = cp.Problem(objective, constraints)
193        problem.solve(solver='SCS', verbose=False)
194    
195        self.alpha = alpha.value
196        self.alpha_star = alpha_star.value
